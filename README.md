# Features
- Add multiple locations and vehicles
- 50% of the money back when returning the vehicle
- Keys and papers have data for the vehicle plate

# Install

--add to your ox_inventory/data/items.lua

``["rental_papers"] = {
	label = "Rental papers",
	weight = 0,
	stack = false,
	close = true,
	description = "Rental papers for a vehicle",
},``
